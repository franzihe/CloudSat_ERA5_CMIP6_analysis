def plt_var_spatial_month(da):
    f, axsm = plt.subplots(nrows=1, ncols=1, subplot_kw={'projection': ccrs.NorthPolarStereo(central_longitude=0.0, globe=None),}, figsize=[3, 3], sharex=True, sharey=True)
    coast = cy.feature.NaturalEarthFeature(category='physical', scale='110m', facecolor='none', name='coastline')

    axsm.set_extent([-180, 180, 90, 45], ccrs.PlateCarree())
    axsm.add_feature(coast, alpha=0.5)
    gl = axsm.gridlines(draw_labels=True)
    gl.top_labels = False
    gl.right_labels = False

    cmap = cm.hawaii_r

    da.groupby('time.month').mean('time', skipna=True).sel(month=1, lat=slice(45,90)).plot(ax=axsm, transform=ccrs.PlateCarree(), cmap=cmap)
